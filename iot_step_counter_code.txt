#include <Wire.h>
#include <Adafruit_MPU6050.h>
#include <Adafruit_Sensor.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

/* OLED */
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

/* MPU6050 */
Adafruit_MPU6050 mpu;

/* Reset button */
#define RESET_BUTTON_PIN 23

/* Step counter variables */
volatile uint16_t stepCount = 0;
float prevMagnitude = 0;

/* Step detection tuning */
#define STEP_THRESHOLD 1.25     // Sensitivity (adjust if needed)
#define STEP_DELAY 300          // ms
unsigned long lastStepTime = 0;

void setup() {
  Serial.begin(115200);
  while (!Serial) {}

  pinMode(RESET_BUTTON_PIN, INPUT_PULLDOWN);

  /* OLED init */
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED failed");
    while (1);
  }

  /* MPU6050 init */
  if (!mpu.begin()) {
    Serial.println("MPU6050 not found");
    while (1);
  }

  mpu.setAccelerometerRange(MPU6050_RANGE_2_G);
  mpu.setGyroRange(MPU6050_RANGE_250_DEG);
  mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);

  display.clearDisplay();
  display.display();
}

void loop() {
  /* Reset button */
  if (digitalRead(RESET_BUTTON_PIN) == HIGH) {
    stepCount = 0;
    Serial.println("Counter reset");
    delay(300);
  }

  /* Read MPU6050 */
  sensors_event_t a, g, temp;
  mpu.getEvent(&a, &g, &temp);

  /* Calculate acceleration magnitude */
  float magnitude = sqrt(
    a.acceleration.x * a.acceleration.x +
    a.acceleration.y * a.acceleration.y +
    a.acceleration.z * a.acceleration.z
  );

  /* Step detection */
  if (magnitude > STEP_THRESHOLD && prevMagnitude <= STEP_THRESHOLD) {
    if (millis() - lastStepTime > STEP_DELAY) {
      stepCount++;
      lastStepTime = millis();
    }
  }
  prevMagnitude = magnitude;

  Serial.print("Steps: ");
  Serial.println(stepCount);

  /* OLED drawing */
  display.clearDisplay();
  display.drawRect(0, 0, SCREEN_WIDTH, SCREEN_HEIGHT, SSD1306_WHITE);

  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(20, 3);
  display.println("MPU6050 STEP COUNTER");
  display.drawLine(0, 15, SCREEN_WIDTH, 15, SSD1306_WHITE);

  display.setTextSize(3);
  display.setCursor(40, 25);
  display.println(stepCount);

  display.setTextSize(1);
  display.setCursor(50, 55);
  display.println("Steps");

  display.display();
  delay(200);
}
